<script src="https://cdn.jsdelivr.net/npm/swiper@11/swiper-bundle.min.js"></script>
<script>
  // Navbar mobile toggle
  const menuBtn = document.getElementById("menu-btn");
  const menu = document.getElementById("menu");
  menuBtn.addEventListener("click", () => menu.classList.toggle("hidden"));

  // Dynamic Sections
  async function loadData(file, containerSelector, type) {
    const res = await fetch(file);
    const data = await res.json();
    const container = document.querySelector(containerSelector);

    data.forEach(item => {
      if(type==='services'){
        const div = document.createElement('div');
        div.className = 'bg-[#1b1b1b] shadow-lg rounded overflow-hidden hover:shadow-2xl transition transform hover:-translate-y-2';
        div.innerHTML = `
          <img src="${item.image}" loading="lazy" class="w-full h-64 object-cover rounded shadow-lg group-hover:opacity-75 transition" alt="${item.title}">
          <div class="p-6">
            <h3 class="text-xl font-semibold text-[#0077cc] mb-2">${item.title}</h3>
            <p class="text-gray-300">${item.description}</p>
          </div>
        `;
        container.appendChild(div);
      } else if(type==='portfolio'){
        const div = document.createElement('div');
        div.className = 'relative group';
        div.innerHTML = `
          <img src="${item.image}" loading="lazy" class="w-full h-64 object-cover rounded shadow-lg group-hover:opacity-75 transition" alt="${item.title}">
          <div class="absolute inset-0 bg-black bg-opacity-50 flex items-center justify-center opacity-0 group-hover:opacity-100 transition">
            <p class="text-white text-lg">${item.title}</p>
          </div>
        `;
        container.appendChild(div);
      } else if(type==='hero'){
        const swiperWrapper = document.querySelector('.swiper-wrapper');
        const slide = document.createElement('div');
        slide.className = 'swiper-slide relative';
        slide.innerHTML = `
          <img src="${item.image}" loading="lazy" class="w-full h-full object-cover" alt="${item.title}">
          <div class="absolute inset-0 bg-black/40 flex flex-col items-center justify-center text-center px-6">
            <h1 class="text-4xl md:text-6xl font-bold mb-4 drop-shadow-lg">${item.title}</h1>
            <p class="text-lg md:text-2xl mb-6 drop-shadow">${item.subtitle}</p>
            <a href="${item.cta}" class="bg-[#0077cc] hover:bg-[#005fa3] px-6 py-3 rounded shadow-lg hover:scale-105 transition">Request a Quote</a>
          </div>
        `;
        swiperWrapper.appendChild(slide);
      }
    });

    // Initialize Swiper after hero images loaded
    if(type==='hero'){
      new Swiper(".swiper", {
        loop: true,
        autoplay: { delay: 4000, disableOnInteraction: false },
        pagination: { el: ".swiper-pagination", clickable: true },
        navigation: { nextEl: ".swiper-button-next", prevEl: ".swiper-button-prev" },
      });
    }
  }

  loadData('data/hero.json', '.swiper-wrapper', 'hero');
  loadData('data/services.json', '#services .grid', 'services');
  loadData('data/portfolio.json', '#portfolio .grid', 'portfolio');
</script>
